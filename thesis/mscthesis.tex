\newcommand{\semanticrule}[2]{%
{
\centering
\def\defaultHypSeparation{\hskip 1pt}
\footnotesize
% \small\scshape
% \small\itshape
\arraycolsep=2.5pt
\begin{minipage}[t]{\textwidth}
\vspace*{-\baselineskip} 
\centering
\setlength{\abovedisplayskip}{0pt}
  \begin{align*}
  \rlap{\hypertarget{#1}{\textsc{#1}}}\phantom{#2} \\[-5pt]
  #2
  \end{align*}
\end{minipage}
}
}
